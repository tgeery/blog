This entry describes a quick setup of a React app with bootstrap.

Install Pre-requisites

npm install react-bootstrap bootstrap

Setup Project

npx create-react-app project
cd project

Edit src/App.js

import React from 'react';
import logo from './logo.svg';
import Jumbotron from 'react-bootstrap/Jumbotron';
import Container from 'react-bootstrap/Container';
import './App.css';

function App() {
  return (
  <Container>
	<nav class="navbar navbar-expand-sm navbar-static-top bg-dark navbar-dark">
	  <form class="navbar-form">
	    <input type="text" class="span1" placeholder="username"/>
	    <input type="text" class="span1" placeholder="password"/>
	    <button type="submit" class="btn" className="statusbutton">Login</button>
	  </form>
	</nav>
	<Jumbotron>
      <h1 className="header">Welcome To React-Bootstrap</h1>
    </Jumbotron>
	<div class="navbar" className="menu">
	  <div class="navbar-inner">
	    <ul class="nav">
	      <li><a className="menuitem" href="#">Surveys</a></li>
	      <li><a className="menuitem" href="#">Predictions</a></li>
	    </ul>
	  </div>
	</div>
	<div class="navbar" className="menusearch">
	  <form class="navbar-search">
	    <input type="text" class="search-query" className="menusearchitem" placeholder="Search" />
	  </form>
	</div>
  </Container>
  );
}

export default App;
