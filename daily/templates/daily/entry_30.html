This entry describes compiling Qt Android on Ubuntu using vlc-android and vlc-qt.

Pre-requisite Install

Compile Qt 5.14 32-bit libraries

cd $HOME/Qt/5.14.1/Src
./configure -platform linux-g++-32 -prefix $HOME/Qt/5.14.1/gcc_32
make
make install
cd $HOME

Clone vlc-android and set environment variables

git clone https://code.videolan.org/videolan/vlc-android.git
export ANDROID_SDK=$HOME/Android/Sdk
export ANDROID_NDK=$ANDROID_SDK/ndk/21.0.6113669
export PATH=$PATH:$ANDROID_SDK/platform-tools:$ANDROID_SDK/tools

Clone vlc-qt

git clone git clone --recursive git://github.com/vlc-qt/vlc-qt.git

Compile for x86_64 (libraries set in PATH)

cd vlc-android
./buildsystem/compile.sh -a x86_64
cd ../vlc-qt
mkdir build
cd build
cmake .. -DQt5Core_DIR=$HOME/Qt/5.14.1/gcc_64/lib/cmake/Qt5Core -DQt5Quick_DIR=$HOME/Qt/5.14.1/gcc_64/lib/cmake/Qt5Quick -DQt5Widgets_DIR=$HOME/Qt/5.14.1/gcc_64/lib/cmake/Qt5Widgets -DQt5QuickTest_DIR=$HOME/Qt/5.14.1/gcc_64/lib/cmake/Qt5QuickTest
make

Compile for armeabi-v7a

cd vlc-android
./buildsystem/compile.sh -a armeabi-v7a
cd ../vlc-qt
mkdir build
cd build
cmake -v .. -DCMAKE_SYSTEM_NAME=Android -DCMAKE_ANDROID_NDK=/home/travis/Android/Sdk/ndk/21.0.6113669 -DCMAKE_TOOLCHAIN_FILE=/home/travis/Android/Sdk/ndk/21.0.6113669/build/cmake/android.toolchain.cmake -DANDROID_PLATFORM=android-21 -DQt5Core_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Core -DQt5Quick_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Quick -DQt5QmlModels_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5QmlModels -DQt5Qml_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Qml -DQt5Network_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Network -DQt5Gui_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Gui -DQt5Widgets_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Widgets -DQt5QuickTest_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5QuickTest -DQt5Test_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Test -DCMAKE_CXX_STANDARD_LIBRARIES="/home/travis/develop/vlc-android/vlc/build-android-arm-linux-androideabi/ndk/libs/armeabi-v7a/libvlc.so /home/travis/Android/Sdk/ndk/21.0.6113669/toolchains/llvm/prebuilt/linux-x86_64/sysroot/usr/lib/arm-linux-androideabi/28/libGLESv3.so"
make
find . -name *.so
./android-build/libs/armeabi-v7a/libVLCQtQml_armeabi-v7a.so
./android-build/libs/armeabi-v7a/libVLCQt_armeabi-v7a.so
./android-build/libs/armeabi-v7a/libVLCQtCore_armeabi-v7a.so
cd ..

Compile for arm64-v8a

cd vlc-android
./buildsystem/compile.sh -a arm64-v8a
cd ../vlc-qt
mkdir build
cd build
cmake .. -DARCH=arm64-v8a -DANDROID_ABI=arm64-v8a -DCMAKE_ANDROID_NDK=/home/travis/Android/Sdk/ndk/21.0.6113669 -DCMAKE_TOOLCHAIN_FILE=/home/travis/Android/Sdk/ndk/21.0.6113669/build/cmake/android.toolchain.cmake -DANDROID_PLATFORM=android-21 -DQt5Core_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Core -DQt5Quick_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Quick -DQt5QmlModels_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5QmlModels -DQt5Qml_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Qml -DQt5Network_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Network -DQt5Gui_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Gui -DQt5Widgets_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Widgets -DQt5QuickTest_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5QuickTest -DQt5Test_DIR=/home/travis/Qt/5.14.1/android/lib/cmake/Qt5Test -DCMAKE_CXX_STANDARD_LIBRARIES="/home/travis/develop/vlc-android/vlc/build-android-aarch64-linux-android/ndk/libs/arm64-v8a/libvlc.so /home/travis/Android/Sdk/ndk/21.0.6113669/toolchains/llvm/prebuilt/linux-x86_64/sysroot/usr/lib/aarch64-linux-android/28/libGLESv3.so"
